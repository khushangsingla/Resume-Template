\pagenumbering{gobble}
% To make a line after heading
\titleformat{\section}{\large\scshape\raggedright}{}{0em}{}

\newcommand{\xfilll}[2][1ex]{
	\dimen0=#2\advance\dimen0 by #1
	\leaders\hrule height \dimen0 depth -#1\hfill
}

% Command \subpoint to use list items
% \subpoint[n]{text} is used for writing text as list item with n(optional) as character/word spacing
\newcommand{\subpoint}[2][0]{\item \textls[#1]{#2}}

% This command is used to write the heading
% \heading{heading}
\newcommand{\heading}[1]{
	\section*{{\LARGE #1}\xfilll[0pt]{0.5pt}}
	\vspace{-8pt}
}

% This command is for point with year at right
% \point[n]{text}{year} where n(optional) is character/word spacing
\newcommand{\point}[3][0]{\item \textls[#1]{#2} \hfill{\sl \small (#3)}}
% Command for project headings
% \projectname{Project}{Duration}{Other Points}{Organization}
\newcommand{\projectname}[4]{
	\noindent\textbf{#1} \hfill{\sl \small #2}\\
	{\it #3} \hfill {\it #4}\\
}

% Use this environment for writing projects name and description
% The arguments are Project name, duration, other points, organization
% Inside the environment, write points using \subpoint command
\newenvironment{project}[4]{
	\projectname{#1}{#2}{#3}{#4}
	\vspace{-17pt}
	\begin{itemize}[itemsep = -0.5 mm, leftmargin=*]
}{
	\end{itemize}
	\vspace{\baselineskip}
	\vspace{-13pt}
}

% Use this environemnt to write sections
% Argument is heading of section
% use \point command inside this environment
\newenvironment{mainsection}[1]{
	\heading{#1}
	\begin{itemize}[itemsep = -0.4 mm, leftmargin=*]
}{
	\end{itemize}
	\vspace{-23pt}
}
